SLIDER DE IMAGEN QUE CAMBIAN AUTOMATICAMENTE
<style>
                        /* Estilo para centrar la imagen */
                        .image-container {
                          text-align: center;
                        }
                    
                        /* Estilo para hacer que las imágenes ocupen todo el ancho de la vista */
                        img {
                          width: 100%;
                          height: 70%;
                        }
                      </style>
                    </head>
                    <body>
                    
                    <div class="image-container">
                      <!-- Imágenes que quieres rotar -->
                      <img src="/assets/img/servicio1.jpg" alt="Imagen 1">
                      <img src="/assets/img/servicio2.jpg" alt="Imagen 2">
                      <img src="/assets/img/servicio3.jpg" alt="Imagen 3">
                    </div>
                    
                    <script>
                      // Obtiene todas las imágenes dentro del contenedor
                      var imagenes = document.querySelectorAll('.image-container img');
                      var indiceActual = 0;
                    
                      // Función para cambiar la imagen automáticamente cada 3 segundos
                      function cambiarImagen() {
                        // Oculta todas las imágenes
                        for (var i = 0; i < imagenes.length; i++) {
                          imagenes[i].style.display = 'none';
                        }
                    
                        // Muestra la siguiente imagen
                        imagenes[indiceActual].style.display = 'block';
                    
                        // Incrementa el índice para apuntar a la siguiente imagen en el próximo cambio
                        indiceActual++;
                    
                        // Reinicia el índice si llega al final del array
                        if (indiceActual === imagenes.length) {
                          indiceActual = 0;
                        }
                      }
                    
                      // Inicia el carrusel
                      setInterval(cambiarImagen, 3000); // Cambia la imagen cada 3 segundos
                    </script>

<div class="slider">
  <!-- Coloca tus imágenes aquí -->
  <img class="slide active" src="imagen1.jpg" alt="Imagen 1">
  <img class="slide" src="imagen2.jpg" alt="Imagen 2">
  <img class="slide" src="imagen3.jpg" alt="Imagen 3">
</div>

<script>
  var slides = document.querySelectorAll('.slide');
  var currentSlide = 0;

  function mostrarSlide(n) {
    slides[currentSlide].classList.remove('active');
    currentSlide = (n + slides.length) % slides.length;
    slides[currentSlide].classList.add('active');
  }

  function avanzarSlide() {
    mostrarSlide(currentSlide + 1);
  }

  setInterval(avanzarSlide, 3000); // Cambiar la imagen cada 3 segundos
</script>
